'use strict'; var BaseClient = require('./BaseClient'); var util = require('util');

Client([options])

Generic HTTP client with helper methods to make requests. Extends BaseClient.

function Client(options) { BaseClient.call(this, options); } util.inherits(Client, BaseClient); module.exports = Client;

Client#request(method, [options])

Make a request. Set method to be any HTTP method, as understood by the configured transport. Returns a Response promise.

For documentation on options see the Request class.

Client.prototype.request = function(method, options) { return this._issueRequest(method, options); };
